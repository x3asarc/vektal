```python
# Phase-plan pattern: (13.2-03), (14-01)
PHASE_PLAN_PATTERN = r'\((\d+\.?\d*)-(\d+)\)'

PHASE_ONLY_PATTERN = r'\((?:phase[- ]?)(\d+\.?\d*)\)'

# Requirement pattern: (CHAT-05), (GRAPH-01)
REQUIREMENT_PATTERN = r'\(([A-Z]+-\d+)\)'
```
