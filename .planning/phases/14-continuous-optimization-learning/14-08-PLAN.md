```python
QUERY_TEMPLATES = {
    # File relationships
    "imports": """
        MATCH (f:File {path: $file_path})-[:IMPORTS]->(imported:File)
        RETURN imported.path, imported.purpose
    """,

    "imported_by": """
        MATCH (f:File {path: $file_path})<-[:IMPORTS]-(importer:File)
        RETURN importer.path, importer.purpose
    """,

    # Semantic similarity (vector search)
    "similar_files": """
        MATCH (f:File {path: $file_path})
        CALL db.index.vector.queryNodes('codebase_embeddings', $limit, f.embedding)
        YIELD node, score
        WHERE score > $threshold
        RETURN node.path, node.purpose, score
        ORDER BY score DESC
    """,

    # Planning context
    "planning_context": """
        MATCH (f:File {path: $file_path})-[:IMPLEMENTS]->(plan:PlanningDoc)
        RETURN plan.path, plan.phase_number, plan.goal
    """,

    "phase_code": """
        MATCH (plan:PlanningDoc {phase_number: $phase})<-[:IMPLEMENTS]-(f:File)
        RETURN f.path, f.purpose
    """,

    # Impact analysis
    "impact_radius": """
        MATCH (f:File {path: $file_path})<-[:IMPORTS*1..3]-(dependent:File)
        RETURN dependent.path, length(path) as depth
        ORDER BY depth
    """,

    # Failure patterns (for auto-improver)
    "similar_failures": """
        MATCH (e:Episode {episode_type: 'oracle_decision'})
        WHERE e.test_result = 'FAIL'
        AND e.root_cause STARTS WITH $root_cause_type
        RETURN e.root_cause, e.suggested_fix, e.phase, e.plan, e.created_at
        ORDER BY e.created_at DESC
        LIMIT $limit
    """,

    # Function lookup
    "function_callers": """
        MATCH (caller:Function)-[:CALLS]->(f:Function {full_name: $function_name})
        RETURN caller.full_name, caller.file_path
    """,

    "function_callees": """
        MATCH (f:Function {full_name: $function_name})-[:CALLS]->(callee:Function)
        RETURN callee.full_name, callee.file_path
    """
}
```

**Implementation:**

   ```python
   # Match query to template by keywords

   if template_match:
       return execute_template(template_match.template, template_match.params)
   ```

   ```python
   if use_natural_language:
       # Generate Cypher from natural language
       cypher = generate_cypher_from_natural_language(query)
       return execute_cypher(cypher)
   ```

```python
#!/usr/bin/env python3
"""
Graph-based Pattern Detector (Phase 14 upgrade)
Queries Neo4j for similar failures in <100ms
Replaces file-based scanner (5-10s)
"""

def find_similar_failures(current_failure: dict) -> list:
    """
    Query graph for similar failures.
    100x faster than file-based version.
    """
    root_cause_type = current_failure.get("root_cause", "unknown").split(":")[0]

    # Use template query
    results = execute_template("similar_failures", {
        "root_cause_type": root_cause_type,
        "limit": 10
    })

    # Format for compatibility with existing orchestrator
    return [
        {
            "root_cause": r["root_cause"],
            "fix": r.get("suggested_fix"),
            "occurrences": 1,  # Count from graph
            "recent_occurrences": [...]
        }
        for r in results
    ]
```
