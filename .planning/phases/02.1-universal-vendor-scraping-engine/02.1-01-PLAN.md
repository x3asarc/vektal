from pydantic import BaseModel, Field
from typing import Optional
from datetime import datetime

class KnownVendor(BaseModel):
    name: str
    sku_pattern: Optional[str] = None
    product_count: int = 0
    last_scrape: Optional[datetime] = None

class CatalogStats(BaseModel):
    total_products: int = 0
    total_vendors: int = 0
    avg_products_per_vendor: float = 0
    most_common_categories: dict[str, int] = Field(default_factory=dict)

class DiscoverySettings(BaseModel):
    require_confirmation_if_confidence_below: float = 0.70
    reject_vendor_if_niche_mismatch: bool = True
    allow_cross_niche_products: bool = False

class StoreProfile(BaseModel):
    store_id: str
    created: datetime = Field(default_factory=datetime.utcnow)
    last_updated: datetime = Field(default_factory=datetime.utcnow)

    niche_primary: str
    niche_sub_niches: list[str] = Field(default_factory=list)
    niche_confidence: float = 0.5

    language: str = "de"
    country: str = "AT"
    vendor_scope: str = "focused"  # focused | flexible | multi_category
    keywords: list[str] = Field(default_factory=list)
    known_vendors: list[KnownVendor] = Field(default_factory=list)

    catalog_stats: CatalogStats = Field(default_factory=CatalogStats)
    discovery_settings: DiscoverySettings = Field(default_factory=DiscoverySettings)

    # Content framework (learned from catalog)
    content_framework: Optional[dict] = None
```

from pathlib import Path
import yaml
from .vendor_schema import VendorConfig
from .store_profile_schema import StoreProfile

def load_vendor_config(yaml_path: Path | str) -> VendorConfig:
    """Load and validate vendor YAML configuration"""
    path = Path(yaml_path)
    if not path.exists():
        raise FileNotFoundError(f"Vendor config not found: {path}")

    with open(path) as f:
        data = yaml.safe_load(f)

    try:
        return VendorConfig.model_validate(data)
    except Exception as e:
        raise ValueError(f"Invalid vendor config {path.name}: {e}")

def load_store_profile(yaml_path: Path | str) -> StoreProfile:
    """Load and validate store profile"""
    path = Path(yaml_path)
    if not path.exists():
        raise FileNotFoundError(f"Store profile not found: {path}")

    with open(path) as f:
        data = yaml.safe_load(f)

    return StoreProfile.model_validate(data)

def save_vendor_config(config: VendorConfig, yaml_path: Path | str) -> None:
    """Save vendor config to YAML"""
    path = Path(yaml_path)
    path.parent.mkdir(parents=True, exist_ok=True)

    with open(path, 'w') as f:
        yaml.dump(
            config.model_dump(exclude_none=True),
            f,
            default_flow_style=False,
            sort_keys=False,
            allow_unicode=True
        )

def save_store_profile(profile: StoreProfile, yaml_path: Path | str) -> None:
    """Save store profile to YAML"""
    path = Path(yaml_path)
    path.parent.mkdir(parents=True, exist_ok=True)

    with open(path, 'w') as f:
        yaml.dump(
            profile.model_dump(exclude_none=True, mode='json'),
            f,
            default_flow_style=False,
            sort_keys=False,
            allow_unicode=True
        )

def list_vendor_configs(config_dir: Path | str = "config/vendors") -> list[Path]:
    """List all vendor YAML files"""
    path = Path(config_dir)
    if not path.exists():
        return []
    return [f for f in path.glob("*.yaml") if not f.name.startswith("_")]
```

import pytest
import re
from src.core.config.vendor_schema import (
    VendorConfig, SKUPattern, VendorURLs, VendorIdentity,
    VendorNiche, ScrapingConfig, Selectors
)
from src.core.config.store_profile_schema import StoreProfile, KnownVendor
from src.core.config import load_vendor_config, save_vendor_config
from pathlib import Path

import yaml


class TestSKUPattern:
    """Test SKU pattern validation"""

    def test_valid_regex_pattern(self):
        """Valid regex compiles successfully"""
        pattern = SKUPattern(
            name="standard",
            regex=r"^R\d{4}[A-Z]?$",
            description="ITD rice paper",
            examples=["R0530", "R0530L"]
        )
        assert pattern.regex == r"^R\d{4}[A-Z]?$"

    def test_invalid_regex_raises_error(self):
        """Invalid regex raises ValueError"""
        with pytest.raises(ValueError, match="Invalid regex"):
            SKUPattern(
                name="bad",
                regex=r"[invalid(",  # Unclosed bracket
                description="Bad pattern",
                examples=[]
            )

    def test_pattern_matches_examples(self):
        """Pattern should match its own examples"""
        pattern = SKUPattern(
            name="test",
            regex=r"^R\d{4}[A-Z]?$",
            description="Test",
            examples=["R0530", "R1234L"]
        )
        compiled = re.compile(pattern.regex)
        for example in pattern.examples:
            assert compiled.match(example), f"Pattern should match example: {example}"


class TestVendorURLs:
    """Test URL template validation"""

    def test_valid_product_template(self):
        """Product URL with {sku} placeholder is valid"""
        urls = VendorURLs(
            product={"template": "https://example.com/products/{sku_lower}"}
        )
        assert "{sku" in urls.product["template"]

    def test_product_template_requires_sku_placeholder(self):
        """Product URL without {sku} placeholder raises error"""
        with pytest.raises(ValueError):
            VendorURLs(
                product={"template": "https://example.com/products/static"}
            )
