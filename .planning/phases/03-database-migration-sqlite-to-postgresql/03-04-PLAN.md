#!/usr/bin/env python
"""

def find_pentart_csv() -> Path:
    """Find the Pentart catalog CSV file."""
    # Common locations
    candidates = [
        Path('data/pentart_catalog.csv'),
        Path('data/csv/pentart_catalog.csv'),
        Path('data/pentart.csv'),
        Path('pentart_catalog.csv'),
    ]

    for path in candidates:
        if path.exists():
            return path

    raise FileNotFoundError(
        f"Pentart CSV not found. Tried: {[str(p) for p in candidates]}"
    )


def import_pentart_catalog(csv_path: Path, user_id: int = 1) -> dict:
    """
    Import Pentart catalog data into PostgreSQL.

    Args:
        csv_path: Path to Pentart CSV file
        user_id: User ID to associate vendor with (default: 1 for initial setup)

    Returns:
        Dict with import statistics
    """
    print(f"Reading CSV: {csv_path}")

    # Read CSV with pandas
    df = pd.read_csv(csv_path)
    print(f"Found {len(df)} rows")

    # Map column names (adjust based on actual CSV headers)
    # Expected columns from SQLite schema: article_number, ean, product_weight
    column_map = {
        'article_number': ['article_number', 'Article Number', 'SKU', 'sku', 'Article_Number'],
        'ean': ['ean', 'EAN', 'barcode', 'Barcode', 'EAN/UPC'],
        'weight': ['product_weight', 'weight', 'Weight', 'weight_grams', 'ProductWeight'],
    }

    # Find actual column names
    sku_col = None
    ean_col = None
    weight_col = None

    for target, candidates in [('sku', column_map['article_number']),
                               ('ean', column_map['ean']),
                               ('weight', column_map['weight'])]:
        for col in candidates:
            if col in df.columns:
                if target == 'sku':
                    sku_col = col
                elif target == 'ean':
                    ean_col = col
                else:
                    weight_col = col
                break

    print(f"Column mapping: sku={sku_col}, ean={ean_col}, weight={weight_col}")

    # Get or create Pentart vendor
    vendor = Vendor.query.filter_by(user_id=user_id, code='PENTART').first()
    if not vendor:
        vendor = Vendor(
            user_id=user_id,
            name='Pentart',
            code='PENTART',
            website_url='https://pentart.eu',
            config_file='pentart.yaml'
        )
        db.session.add(vendor)
        db.session.flush()  # Get vendor.id
        print(f"Created Pentart vendor (id={vendor.id})")
    else:
        # Clear existing items for re-import
        VendorCatalogItem.query.filter_by(vendor_id=vendor.id).delete()
        print(f"Using existing Pentart vendor (id={vendor.id}), cleared {vendor.catalog_item_count or 0} items")

    # Prepare bulk insert
    items = []
    skipped = 0

    for idx, row in df.iterrows():
        sku = str(row[sku_col]).strip() if sku_col and pd.notna(row.get(sku_col)) else None
        ean = str(row[ean_col]).strip() if ean_col and pd.notna(row.get(ean_col)) else None
        weight = None
        if weight_col and pd.notna(row.get(weight_col)):
            try:
                weight = int(float(row[weight_col]))
            except (ValueError, TypeError):
                pass

        # Skip rows with no identifiers
        if not sku and not ean:
            skipped += 1
            continue

        items.append({
            'vendor_id': vendor.id,
            'sku': sku,
            'barcode': ean,
            'weight_grams': weight,
            'source_row_number': idx + 1,
        })

    # Bulk insert for performance
    if items:
        db.session.bulk_insert_mappings(VendorCatalogItem, items)

    # Update vendor stats
    vendor.catalog_item_count = len(items)
    from datetime import datetime, timezone
    vendor.last_import_at = datetime.now(timezone.utc)

    db.session.commit()

    stats = {
        'total_rows': len(df),
        'imported': len(items),
        'skipped': skipped,
        'vendor_id': vendor.id,
    }

    print(f"\nImport complete:")
    print(f"  Total rows: {stats['total_rows']}")
    print(f"  Imported: {stats['imported']}")
    print(f"  Skipped (no identifiers): {stats['skipped']}")

    return stats


def main():
    parser = argparse.ArgumentParser(description='Import Pentart catalog to PostgreSQL')
    parser.add_argument('--csv-path', type=Path, help='Path to Pentart CSV file')
    parser.add_argument('--user-id', type=int, default=1, help='User ID to associate vendor with')
    args = parser.parse_args()

    # Create Flask app context
    app = create_app()

    with app.app_context():
        # Ensure tables exist
        db.create_all()

        # Ensure user exists (for foreign key)
        user = User.query.get(args.user_id)
        if not user:
            user = User(id=args.user_id, email='admin@localhost')
            db.session.add(user)
            db.session.commit()
            print(f"Created admin user (id={user.id})")

        # Find and import CSV
        csv_path = args.csv_path or find_pentart_csv()
        stats = import_pentart_catalog(csv_path, args.user_id)

        # Verify import
        count = VendorCatalogItem.query.filter_by(vendor_id=stats['vendor_id']).count()
        print(f"\nVerification: {count} items in database")


if __name__ == '__main__':
    main()

def verify_webhook(data, hmac_header):
    """Verify Shopify webhook signature."""
    if not hmac_header:
        return False
    calculated_hmac = hmac.new(
        SHOPIFY_API_SECRET.encode('utf-8'),
        data.encode('utf-8'),
        hashlib.sha256
    ).hexdigest()
    return hmac.compare_digest(calculated_hmac, hmac_header)
