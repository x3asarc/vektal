```python
#!/usr/bin/env python
"""
Import Pentart catalog data into PostgreSQL.

IMPORTANT: This is NOT a migration from SQLite.
Per CONTEXT.md:
- Production schema designed from requirements, not SQLite structure
- Only importing 3 columns from original 10-column CSV:
  * barcode (EAN)
  * SKU (article_number)
  * weight (product_weight)
- Hungarian titles and other columns are NOT imported

This is INITIAL VENDOR CATALOG DATA for the fresh PostgreSQL schema.

Usage:
    python scripts/import_pentart.py [--csv-path PATH]

    # With Docker container running:
    docker compose exec backend python scripts/import_pentart.py
"""
import os
import sys
import argparse
from pathlib import Path

# Add project root to path
sys.path.insert(0, str(Path(__file__).parent.parent))

import pandas as pd
from flask import Flask

from src.database import create_app, db
from src.models import User, Vendor, VendorCatalogItem


def find_pentart_csv() -> Path:
    """Find the Pentart catalog CSV file."""
    # Common locations
    candidates = [
        Path('data/pentart_catalog.csv'),
        Path('data/csv/pentart_catalog.csv'),
        Path('data/pentart.csv'),
        Path('pentart_catalog.csv'),
    ]

    for path in candidates:
        if path.exists():
            return path

    raise FileNotFoundError(
        f"Pentart CSV not found. Tried: {[str(p) for p in candidates]}"
    )


def import_pentart_catalog(csv_path: Path, user_id: int = 1) -> dict:
    """
    Import Pentart catalog data into PostgreSQL.

    This imports INITIAL VENDOR CATALOG DATA - not a schema migration.
    Per CONTEXT.md: Only 3 columns are useful (barcode, SKU, weight).

    Args:
        csv_path: Path to Pentart CSV file
        user_id: User ID to associate vendor with (default: 1 for initial setup)

    Returns:
        Dict with import statistics
    """
    print(f"Reading CSV: {csv_path}")
    print("NOTE: Importing as INITIAL VENDOR CATALOG (not SQLite migration)")

    # Read CSV with pandas
    df = pd.read_csv(csv_path)
    print(f"Found {len(df)} rows")

    # Map column names - only the 3 useful columns per CONTEXT.md
    # (barcode/EAN, SKU/article_number, weight)
    column_map = {
        'article_number': ['article_number', 'Article Number', 'SKU', 'sku', 'Article_Number'],
        'ean': ['ean', 'EAN', 'barcode', 'Barcode', 'EAN/UPC'],
        'weight': ['product_weight', 'weight', 'Weight', 'weight_grams', 'ProductWeight'],
    }

    # Find actual column names
    sku_col = None
    ean_col = None
    weight_col = None

    for target, candidates in [('sku', column_map['article_number']),
                               ('ean', column_map['ean']),
                               ('weight', column_map['weight'])]:
        for col in candidates:
            if col in df.columns:
                if target == 'sku':
                    sku_col = col
                elif target == 'ean':
                    ean_col = col
                else:
                    weight_col = col
                break

    print(f"Column mapping: sku={sku_col}, ean={ean_col}, weight={weight_col}")
    print("(Only these 3 columns imported - other columns ignored per CONTEXT.md)")

    # Get or create Pentart vendor
    vendor = Vendor.query.filter_by(user_id=user_id, code='PENTART').first()
    if not vendor:
        vendor = Vendor(
            user_id=user_id,
            name='Pentart',
            code='PENTART',
            website_url='https://pentart.eu',
            config_file='pentart.yaml'
        )
        db.session.add(vendor)
        db.session.flush()  # Get vendor.id
        print(f"Created Pentart vendor (id={vendor.id})")
    else:
        # Clear existing items for re-import
        VendorCatalogItem.query.filter_by(vendor_id=vendor.id).delete()
        print(f"Using existing Pentart vendor (id={vendor.id}), cleared {vendor.catalog_item_count or 0} items")

    # Prepare bulk insert
    items = []
    skipped = 0

    for idx, row in df.iterrows():
        sku = str(row[sku_col]).strip() if sku_col and pd.notna(row.get(sku_col)) else None
        ean = str(row[ean_col]).strip() if ean_col and pd.notna(row.get(ean_col)) else None
        weight = None
        if weight_col and pd.notna(row.get(weight_col)):
            try:
                weight = int(float(row[weight_col]))
            except (ValueError, TypeError):
                pass

        # Skip rows with no identifiers
        if not sku and not ean:
            skipped += 1
            continue

        items.append({
            'vendor_id': vendor.id,
            'sku': sku,
            'barcode': ean,
            'weight_grams': weight,
            'source_row_number': idx + 1,
            # NOTE: title NOT imported - was Hungarian, not applicable per CONTEXT.md
        })

    # Bulk insert for performance
    if items:
        db.session.bulk_insert_mappings(VendorCatalogItem, items)

    # Update vendor stats
    vendor.catalog_item_count = len(items)
    from datetime import datetime, timezone
    vendor.last_import_at = datetime.now(timezone.utc)

    db.session.commit()

    stats = {
        'total_rows': len(df),
        'imported': len(items),
        'skipped': skipped,
        'vendor_id': vendor.id,
    }

    print(f"\nInitial catalog import complete:")
    print(f"  Total rows: {stats['total_rows']}")
    print(f"  Imported: {stats['imported']} (barcode, SKU, weight only)")
    print(f"  Skipped (no identifiers): {stats['skipped']}")

    return stats


def main():
    parser = argparse.ArgumentParser(
        description='Import Pentart catalog to PostgreSQL (initial vendor catalog data)'
    )
    parser.add_argument('--csv-path', type=Path, help='Path to Pentart CSV file')
    parser.add_argument('--user-id', type=int, default=1, help='User ID to associate vendor with')
    args = parser.parse_args()

    # Create Flask app context
    app = create_app()

    with app.app_context():
        # Ensure tables exist
        db.create_all()

        # Ensure user exists (for foreign key)
        user = User.query.get(args.user_id)
        if not user:
            user = User(id=args.user_id, email='admin@localhost')
            db.session.add(user)
            db.session.commit()
            print(f"Created admin user (id={user.id})")

        # Find and import CSV
        csv_path = args.csv_path or find_pentart_csv()
        stats = import_pentart_catalog(csv_path, args.user_id)

        # Verify import
        count = VendorCatalogItem.query.filter_by(vendor_id=stats['vendor_id']).count()
        print(f"\nVerification: {count} items in database")


if __name__ == '__main__':
    main()
```

```dockerfile
# Run migrations then start gunicorn
# flask db upgrade applies pending migrations
# gunicorn serves the application
CMD ["sh", "-c", "flask db upgrade && gunicorn --config gunicorn.conf.py src.app:app"]
```

Also add FLASK_APP environment variable:
```dockerfile
ENV FLASK_APP=src/app_factory.py
```

This ensures:
1. Database migrations run automatically on container startup
2. App won't start if migrations fail (immediate feedback)
3. Multiple container replicas won't conflict (migrations are idempotent)
