def get_current_user():
    """Get or create user for current session."""
    shop = session.get('shop')
    if not shop:
        return None

    # Find user by shop domain
    store = ShopifyStore.query.filter_by(store_url=shop).first()
    if store:
        return store.user

    # Create user if not exists (temporary - Phase 4 will handle properly)
    user = User(email=f"{shop.replace('.myshopify.com', '')}@placeholder")
    db.session.add(user)
    db.session.flush()

    store = ShopifyStore(
        user_id=user.id,
        store_url=shop,
        api_version=SHOPIFY_API_VERSION
    )
    db.session.add(store)
    db.session.commit()

    return user

def get_jobs():
    user = get_current_user()

    jobs = Job.query.filter_by(user_id=user.id).order_by(Job.created_at.desc()).limit(50).all()

    return jsonify({'jobs': [

            'id': j.id,

            'processed': j.processed_items,

            'updated_count': j.updated_count,
            'created_at': j.created_at.isoformat() if j.created_at else None,

    ]})
