```python
"""Shopify store model with encrypted OAuth token storage."""
from src.models import db, TimestampMixin


class ShopifyStore(db.Model, TimestampMixin):
    """User's connected Shopify store."""
    __tablename__ = 'shopify_stores'

    id = db.Column(db.Integer, primary_key=True)
    user_id = db.Column(db.Integer, db.ForeignKey('users.id'), unique=True, nullable=False)

    # Store identification
    store_url = db.Column(db.String(255), nullable=False)  # xxx.myshopify.com
    shop_name = db.Column(db.String(255))  # Display name

    # OAuth credentials - ENCRYPTED at rest using Fernet
    # Use set_access_token() and get_access_token() methods
    access_token_encrypted = db.Column(db.LargeBinary)

    # API configuration
    api_version = db.Column(db.String(20), default='2024-01')
    scopes = db.Column(db.String(500))  # Granted OAuth scopes

    # Relationships
    products = db.relationship('Product', backref='store', lazy='dynamic')

    def set_access_token(self, token: str) -> None:
        """
        Encrypt and store the access token.

        Args:
            token: Plaintext OAuth access token

        Note: Requires src/core/encryption.py to exist (Plan 03).
              Import is deferred to method call to avoid circular imports
              and allow model definition before encryption module exists.
        """
        if not token:
            self.access_token_encrypted = None
            return
        # Deferred import - encryption.py created in Plan 03
        from src.core.encryption import encrypt_token
        self.access_token_encrypted = encrypt_token(token)

    def get_access_token(self) -> str | None:
        """
        Decrypt and return the access token.

        Returns:
            Decrypted token string, or None if not set or decryption fails

        Note: Requires src/core/encryption.py to exist (Plan 03).
        """
        if not self.access_token_encrypted:
            return None
        # Deferred import - encryption.py created in Plan 03
        from src.core.encryption import decrypt_token
        return decrypt_token(self.access_token_encrypted)

    def __repr__(self):
        return f'<ShopifyStore {self.store_url}>'


class ShopifyCredential(db.Model, TimestampMixin):
    """
    Additional encrypted credentials for Shopify integrations.
    Separate from ShopifyStore to support multiple credential types.
    """
    __tablename__ = 'shopify_credentials'

    id = db.Column(db.Integer, primary_key=True)
    store_id = db.Column(db.Integer, db.ForeignKey('shopify_stores.id'), nullable=False, index=True)

    credential_type = db.Column(db.String(50), nullable=False)  # 'api_key', 'webhook_secret'
    credential_encrypted = db.Column(db.LargeBinary, nullable=False)

    # Composite unique constraint
    __table_args__ = (
        db.UniqueConstraint('store_id', 'credential_type', name='uq_store_credential_type'),
    )

    def set_credential(self, value: str) -> None:
        """Encrypt and store the credential."""
        from src.core.encryption import encrypt_token
        self.credential_encrypted = encrypt_token(value)

    def get_credential(self) -> str | None:
        """Decrypt and return the credential."""
        if not self.credential_encrypted:
            return None
        from src.core.encryption import decrypt_token
        return decrypt_token(self.credential_encrypted)

    def __repr__(self):
        return f'<ShopifyCredential {self.credential_type} for store {self.store_id}>'
```

Key design decisions:
- User has one-to-one with ShopifyStore (v1.0 requirement)
- OAuth token stored as encrypted bytes (LargeBinary)
- ShopifyStore has set_access_token() and get_access_token() methods that use deferred imports to src/core/encryption.py (created in Plan 03)
- ShopifyCredential for additional secrets (API keys, webhook secrets)
- Index on user_id foreign key for efficient JOINs

```python
from src.models import db, TimestampMixin


class Vendor(db.Model, TimestampMixin):
    """Vendor/supplier whose catalog has been imported."""
    __tablename__ = 'vendors'

    id = db.Column(db.Integer, primary_key=True)
    user_id = db.Column(db.Integer, db.ForeignKey('users.id'), nullable=False, index=True)

    # Vendor identification
    name = db.Column(db.String(255), nullable=False)
    code = db.Column(db.String(50), index=True)  # Short code like 'PENTART', 'KREUL'
    website_url = db.Column(db.String(500))

    # YAML config reference (from Phase 2.1)
    config_file = db.Column(db.String(255))  # e.g., 'pentart.yaml'

    # Catalog metadata
    last_import_at = db.Column(db.DateTime(timezone=True))
    catalog_item_count = db.Column(db.Integer, default=0)

    # Relationships
    catalog_items = db.relationship('VendorCatalogItem', backref='vendor', lazy='dynamic',
                                    cascade='all, delete-orphan')

    # User + code should be unique
    __table_args__ = (
        db.UniqueConstraint('user_id', 'code', name='uq_user_vendor_code'),
    )

    def __repr__(self):
        return f'<Vendor {self.code or self.name}>'


class VendorCatalogItem(db.Model, TimestampMixin):
    """
    Parsed vendor catalog data (from CSV imports).
    Supports fast SQL searching across vendor catalogs.
    """
    __tablename__ = 'vendor_catalog_items'

    id = db.Column(db.Integer, primary_key=True)
    vendor_id = db.Column(db.Integer, db.ForeignKey('vendors.id'), nullable=False, index=True)

    # Core identifiers - indexed for fast lookup
    sku = db.Column(db.String(100), index=True)
    barcode = db.Column(db.String(50), index=True)  # EAN/UPC

    # Product data (sparse - only populate what exists in source)
    title = db.Column(db.String(500))
    description = db.Column(db.Text)
    price = db.Column(db.Numeric(10, 2))
    currency = db.Column(db.String(3), default='EUR')
    weight_grams = db.Column(db.Integer)

    # Source tracking
    source_row_number = db.Column(db.Integer)  # Row in original CSV
    raw_data = db.Column(db.JSON)  # Original row data for reference

    # Composite index for fast catalog search (vendor + sku + barcode)
    __table_args__ = (
        db.Index('ix_vendor_catalog_lookup', 'vendor_id', 'sku', 'barcode'),
    )

    def __repr__(self):
        return f'<VendorCatalogItem {self.sku or self.barcode}>'
```

Key design decisions:
- VendorCatalogItem stores parsed CSV data for fast SQL searching (CONTEXT.md requirement)
- Sparse columns - only populate what exists in source CSV
- Composite index on vendor_id + sku + barcode for efficient lookups
- raw_data JSON column preserves original data for debugging
- Cascade delete removes catalog items when vendor is deleted

```python
"""Job model - background task tracking for Celery tasks."""
from src.models import db, TimestampMixin


class Job(db.Model, TimestampMixin):
    """Background job tracking (Celery task metadata)."""
    __tablename__ = 'jobs'

    id = db.Column(db.Integer, primary_key=True)
    user_id = db.Column(db.Integer, db.ForeignKey('users.id'), nullable=False, index=True)

    # Celery task reference
    celery_task_id = db.Column(db.String(50), unique=True, index=True)

    # Job classification
    job_type = db.Column(db.String(50), nullable=False)  # scrape, enrich, sync, import
    status = db.Column(db.String(20), default='pending', index=True)  # pending, running, completed, failed, cancelled

    # Progress tracking
    total_items = db.Column(db.Integer)
    processed_items = db.Column(db.Integer, default=0)
    successful_items = db.Column(db.Integer, default=0)
    failed_items = db.Column(db.Integer, default=0)

    # Job-specific counters (for scraping jobs)
    created_count = db.Column(db.Integer, default=0)
    updated_count = db.Column(db.Integer, default=0)

    # Input data
    input_file = db.Column(db.String(255))  # CSV filename
    input_params = db.Column(db.JSON)  # Job parameters

    # Results
    result_data = db.Column(db.JSON)  # PostgreSQL JSONB
    error_message = db.Column(db.Text)

    # Timing
    started_at = db.Column(db.DateTime(timezone=True))
    completed_at = db.Column(db.DateTime(timezone=True))

    # Relationships
    results = db.relationship('JobResult', backref='job', lazy='dynamic',
                              cascade='all, delete-orphan')

    def __repr__(self):
        return f'<Job {self.id} {self.job_type} {self.status}>'

    @property
    def progress_percent(self) -> int:
        """Calculate progress percentage."""
        if not self.total_items:
            return 0
        return int((self.processed_items / self.total_items) * 100)


class JobResult(db.Model, TimestampMixin):
    """Individual result within a job (e.g., per-product result)."""
    __tablename__ = 'job_results'

    id = db.Column(db.Integer, primary_key=True)
    job_id = db.Column(db.Integer, db.ForeignKey('jobs.id'), nullable=False, index=True)

    # Item identification
    sku = db.Column(db.String(100), index=True)
    handle = db.Column(db.String(255))
    vendor = db.Column(db.String(255))

    # Result status
    status = db.Column(db.String(50))  # created, updated, skipped, failed, not_found
    error_message = db.Column(db.Text)

    # Outcome data
    product_id = db.Column(db.String(50))  # Shopify product GID
    variant_id = db.Column(db.String(50))  # Shopify variant GID
    image_url = db.Column(db.String(1000))
    price = db.Column(db.Numeric(10, 2))

    # Flags
    product_created = db.Column(db.Boolean, default=False)
    inventory_set = db.Column(db.Boolean, default=False)
    inventory_quantity = db.Column(db.Integer)

    def __repr__(self):
        return f'<JobResult {self.sku} {self.status}>'
```
