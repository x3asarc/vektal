```bash
#!/bin/bash
# Database Backup Script
# Creates compressed pg_dump backup in custom format
#
# Usage:
#   ./scripts/backup_db.sh                    # Backup to ./backups/
#   BACKUP_DIR=/mnt/backups ./scripts/backup_db.sh  # Custom location
#
# Restore time target: Under 5 minutes

set -e

# Configuration (override via environment)
BACKUP_DIR="${BACKUP_DIR:-./backups}"
DB_CONTAINER="${DB_CONTAINER:-shopifyscrapingscript-db-1}"
DB_NAME="${DB_NAME:-shopify_platform}"
DB_USER="${DB_USER:-admin}"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
BACKUP_FILE="${BACKUP_DIR}/backup_${TIMESTAMP}.dump"
RETENTION_COUNT="${RETENTION_COUNT:-5}"

echo "=== Database Backup ==="
echo "Container: $DB_CONTAINER"
echo "Database: $DB_NAME"
echo "Output: $BACKUP_FILE"

# Create backup directory
mkdir -p "$BACKUP_DIR"

# Create backup with custom format and gzip compression
# -Fc: custom format (allows pg_restore selective restore)
# -Z 6: compression level 6 (good balance of speed vs size)
echo "Creating backup..."
docker exec "$DB_CONTAINER" pg_dump \
    -U "$DB_USER" \
    -Fc \
    -Z 6 \
    "$DB_NAME" > "$BACKUP_FILE"

# Show backup info
BACKUP_SIZE=$(ls -lh "$BACKUP_FILE" | awk '{print $5}')
echo "Backup created: $BACKUP_FILE ($BACKUP_SIZE)"

# Cleanup old backups (keep last N)
if [ "$RETENTION_COUNT" -gt 0 ]; then
    echo "Cleaning up old backups (keeping last $RETENTION_COUNT)..."
    cd "$BACKUP_DIR" && ls -t backup_*.dump 2>/dev/null | tail -n +$((RETENTION_COUNT + 1)) | xargs -r rm --
fi

# List current backups
echo ""
echo "Current backups:"
ls -lh "$BACKUP_DIR"/backup_*.dump 2>/dev/null || echo "  (none)"

echo ""
echo "=== Backup Complete ==="
```

Create scripts/restore_db.sh:
```bash
#!/bin/bash
# Database Restore Script
# Restores from pg_dump custom format backup
#
# Usage:
#   ./scripts/restore_db.sh backups/backup_20260208_120000.dump
#
# WARNING: This replaces ALL data in the database!

set -e

BACKUP_FILE="$1"
DB_CONTAINER="${DB_CONTAINER:-shopifyscrapingscript-db-1}"
DB_NAME="${DB_NAME:-shopify_platform}"
DB_USER="${DB_USER:-admin}"

if [ -z "$BACKUP_FILE" ]; then
    echo "Usage: $0 <backup_file.dump>"
    echo ""
    echo "Available backups:"
    ls -lh ./backups/backup_*.dump 2>/dev/null || echo "  (none found in ./backups/)"
    exit 1
fi

if [ ! -f "$BACKUP_FILE" ]; then
    echo "Error: Backup file not found: $BACKUP_FILE"
    exit 1
fi

BACKUP_SIZE=$(ls -lh "$BACKUP_FILE" | awk '{print $5}')

echo "=== Database Restore ==="
echo "Backup file: $BACKUP_FILE ($BACKUP_SIZE)"
echo "Container: $DB_CONTAINER"
echo "Database: $DB_NAME"
echo ""
echo "WARNING: This will REPLACE ALL DATA in $DB_NAME!"
echo ""
read -p "Continue? (y/N) " -n 1 -r
echo

if [[ ! $REPLY =~ ^[Yy]$ ]]; then
    echo "Restore cancelled."
    exit 0
fi

# Measure restore time
START_TIME=$(date +%s)

echo "Restoring..."

# Restore from custom format backup
# -c: clean (drop objects before recreating)
# -d: target database
cat "$BACKUP_FILE" | docker exec -i "$DB_CONTAINER" pg_restore \
    -U "$DB_USER" \
    -c \
    -d "$DB_NAME" \
    --if-exists \
    2>&1 || true  # pg_restore may return non-zero for warnings

END_TIME=$(date +%s)
DURATION=$((END_TIME - START_TIME))

echo ""
echo "=== Restore Complete ==="
echo "Duration: ${DURATION} seconds"
echo ""
echo "Verify with: docker exec $DB_CONTAINER psql -U $DB_USER -d $DB_NAME -c '\\dt'"
```

```python
"""
Encryption helpers for secure credential storage.
Uses Fernet symmetric encryption for OAuth tokens and API keys.

The encryption key should be stored in:
1. Docker secret file: /run/secrets/encryption_key (production)
2. Environment variable: ENCRYPTION_KEY (development)

Key rotation: Generate new key, decrypt with old, encrypt with new.
"""
import os
import base64
from typing import Optional
from cryptography.fernet import Fernet, InvalidToken

from src.core.secrets import get_secret


def get_encryption_key() -> bytes:
    """
    Get the Fernet encryption key from secrets or environment.

    Returns:
        32-byte encryption key

    Raises:
        ValueError: If key is not configured or invalid
    """
    key = get_secret('ENCRYPTION_KEY')

    if not key:
        # For development, generate a key if not set
        # WARNING: This means encrypted data won't persist across restarts!
        import warnings
        warnings.warn(
            "ENCRYPTION_KEY not set. Using ephemeral key - encrypted data will be lost on restart!",
            UserWarning
        )
        return Fernet.generate_key()

    # Fernet requires URL-safe base64-encoded 32-byte key
    try:
        # If key is already valid Fernet key, use it
        Fernet(key.encode() if isinstance(key, str) else key)
        return key.encode() if isinstance(key, str) else key
    except Exception:
        # If not valid Fernet format, derive from it
        # This allows using any string as key (will be hashed)
        import hashlib
        key_bytes = key.encode() if isinstance(key, str) else key
        derived = hashlib.sha256(key_bytes).digest()
        return base64.urlsafe_b64encode(derived)


def encrypt_token(plaintext: str) -> bytes:
    """
    Encrypt a token or credential for storage.

    Args:
        plaintext: The token to encrypt

    Returns:
        Encrypted bytes suitable for LargeBinary column
    """
    if not plaintext:
        return b''

    key = get_encryption_key()
    f = Fernet(key)
    return f.encrypt(plaintext.encode('utf-8'))


def decrypt_token(ciphertext: bytes) -> Optional[str]:
    """
    Decrypt a stored token.

    Args:
        ciphertext: The encrypted bytes from database

    Returns:
        Decrypted token string, or None if decryption fails

    Note:
        Returns None (not raises) on failure to handle migration gracefully
    """
    if not ciphertext:
        return None

    try:
        key = get_encryption_key()
        f = Fernet(key)
        return f.decrypt(ciphertext).decode('utf-8')
    except InvalidToken:
        # Key mismatch or corrupted data
        return None
    except Exception:
        return None


def generate_encryption_key() -> str:
    """
    Generate a new Fernet encryption key.

    Usage:
        python -c "from src.core.encryption import generate_encryption_key; print(generate_encryption_key())"

    Returns:
        URL-safe base64-encoded 32-byte key
    """
    return Fernet.generate_key().decode('utf-8')


# Convenience aliases
encrypt = encrypt_token
decrypt = decrypt_token
```
