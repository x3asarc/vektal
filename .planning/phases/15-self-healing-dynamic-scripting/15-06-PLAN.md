import pytest
from src.graph.sentry_ingestor import normalize_sentry_issue
from src.graph.orchestrate_healers import orchestrate_remediation

class TestSentryIntegration:
    """Test Phase 14.3 Sentry ingestion with Phase 15 remediation."""

    def test_infrastructure_failure_flow(self):
        """Redis connection error → infrastructure category → Redis remediator."""
        sentry_issue = {
            'id': 'SENTRY-123',
            'title': 'ConnectionError',
            'culprit': 'src/core/redis.py',
            'type': 'error',
            'metadata': {
                'type': 'ConnectionError',
                'value': 'redis connection refused'
            },
            'entries': [{
                'type': 'exception',
                'data': {
                    'values': [{
                        'stacktrace': {
                            'frames': [
                                {'filename': 'src/core/redis.py', 'lineno': 42}
                            ]
                        }
                    }]
                }
            }]
        }

        result = orchestrate_remediation(sentry_issue)
        assert result['status'] in ['remediated', 'manual_required']
        assert result['category'] == 'infrastructure'

    def test_code_failure_flow(self):
        """ImportError → code category → LLM remediator."""
        sentry_issue = {
            'id': 'SENTRY-456',
            'title': 'ImportError',
            'culprit': 'src/tasks/enrichment.py',
            'type': 'error',
            'metadata': {
                'type': 'ImportError',
                'value': 'No module named missing_lib'
            },
            'entries': [{'type': 'exception', 'data': {'values': [{}]}}]
        }

        result = orchestrate_remediation(sentry_issue)
        assert result['category'] == 'code'

    def test_config_failure_flow(self):
        """Timeout → config category → config remediator."""
        sentry_issue = {
            'id': 'SENTRY-789',
            'title': 'TimeoutError',
            'culprit': 'src/tasks/enrichment.py',
            'type': 'error',
            'metadata': {
                'type': 'TimeoutError',
                'value': 'Task execution timeout'
            },
            'entries': [{'type': 'exception', 'data': {'values': [{}]}}]
        }

        result = orchestrate_remediation(sentry_issue)
        assert result['category'] == 'config'

    def test_normalization_integration(self):
        """Verify Phase 14.3 normalization works with classifier."""
        raw_issue = {
            'id': 'SENTRY-999',
            'title': 'Error',
            'culprit': 'src/test.py',
            'metadata': {'type': 'RuntimeError', 'value': 'test error'},
            'entries': []
        }

        normalized = normalize_sentry_issue(raw_issue)
        assert 'error_type' in normalized
        assert 'error_message' in normalized
        assert 'affected_module' in normalized

        'id': 'TEST-CODE',

        'culprit': 'src/tasks/enrichment.py',
        'metadata': {'type': 'ImportError', 'value': 'No module named test'},
        'entries': []
    }
    result = orchestrate_remediation(issue)
    print(json.dumps(result, indent=2))

if __name__ == '__main__':
    print("Testing infrastructure failure...")
    test_infrastructure()
    print("\nTesting code failure...")
    test_code()
