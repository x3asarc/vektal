```python
class SessionPrimer:
    """Loads and compresses session context for LLM consumption."""

    def __init__(self, memory_loader: MemoryLoader):
        self.memory_loader = memory_loader

    def load_session_context(self, failure_context: Optional[str] = None) -> str:
        """Load compressed session context as YAML string.

        Returns YAML string suitable for LLM prompt injection:
        - 5 most recent git commits (compressed)
        - Current phase and plan
        - Roadmap context (milestone, goal, next phase)
        - 3-5 relevant remedy templates (if failure_context provided)

        Target: <1200 tokens (10x reduction from raw ~12000 tokens)
        """
        # Load raw data
        commits = self.memory_loader.load_recent_commits(limit=5)
        phase_context = self.memory_loader.load_current_phase()
        roadmap = self.memory_loader.load_roadmap_summary()
        remedies = []
        if failure_context:
            remedies = self.memory_loader.load_relevant_remedies(
                failure_context, limit=5
            )

        # Compress to YAML
        return self._compress_to_yaml({
            'session': phase_context,
            'recent_commits': self._compress_commits(commits),
            'roadmap_context': self._compress_roadmap(roadmap),
            'relevant_remedies': self._compress_remedies(remedies)
        })

    def _compress_commits(self, commits: List[GitCommit]) -> List[Dict]:
        """Compress commit data: hash, message, files, impact."""
        return [
            {
                'hash': c.sha[:7],  # Short hash
                'msg': self._truncate(c.message, 60),
                'files': [self._basename(f) for f in c.changed_files[:3]],
                'impact': self._infer_impact_area(c.changed_files)
            }
            for c in commits
        ]

    def _compress_roadmap(self, roadmap: Dict) -> Dict:
        """Extract essential roadmap info only."""
        return {
            'milestone': roadmap.get('current_milestone'),
            'current_phase': roadmap.get('current_phase'),
            'goal': self._truncate(roadmap.get('goal', ''), 80),
            'completed': roadmap.get('phases_complete', []),
            'next': roadmap.get('next_phase')
        }

    def _compress_remedies(self, remedies: List[RemedyTemplate]) -> List[Dict]:
        """Compress remedy template data."""
        return [
            {
                'fp': r.fingerprint,
                'fix': self._truncate(r.description, 50),
                'conf': float(r.confidence),
                'apps': r.application_count
            }
            for r in remedies
        ]

    def _compress_to_yaml(self, data: Dict) -> str:
        """Convert to compact YAML string."""
        import yaml
        return yaml.dump(data, default_flow_style=False, width=80)

    def _truncate(self, text: str, max_len: int) -> str:
        """Truncate text with ellipsis."""
        return text[:max_len] + '...' if len(text) > max_len else text

    def _basename(self, path: str) -> str:
        """Extract filename from path."""
        return path.split('/')[-1]

    def _infer_impact_area(self, files: List[str]) -> str:
        """Infer area of impact from changed files."""
        if any('graph' in f for f in files):
            return 'graph'
        if any('api' in f for f in files):
            return 'api'
        if any('models' in f for f in files):
            return 'database'
        return 'core'
```

from typing import List, Dict, Optional
from src.core.graphiti_client import get_graphiti_client
from src.models.remedy_templates import RemedyTemplate
from datetime import datetime, timedelta

class MemoryLoader:
    """Loads session context from Neo4j graph and PostgreSQL cache."""

    def __init__(self):
        self.graph_client = get_graphiti_client()

    def load_recent_commits(self, limit: int = 5) -> List[Dict]:
        """Load recent git commits from graph.

        Query Neo4j for Commit nodes ordered by timestamp DESC.
        Returns: List of {sha, message, timestamp, changed_files}
        """
        if not self.graph_client:
            return self._fallback_git_log(limit)

        query = '''
        MATCH (c:Commit)
        RETURN c.sha AS sha,
               c.message AS message,
               c.timestamp AS timestamp,
               c.changed_files AS changed_files
        ORDER BY c.timestamp DESC
        LIMIT $limit
        '''
        result = self.graph_client.execute_query(query, {'limit': limit})
        return [dict(record) for record in result]

    def load_current_phase(self) -> Dict:
        """Load current phase and plan from STATE.md or graph."""
        # Parse .planning/STATE.md for current phase
        import re
        with open('.planning/STATE.md') as f:
            content = f.read()
            phase_match = re.search(r'Phase (\d+\.?\d*):', content)
            if phase_match:
                return {
                    'current_phase': phase_match.group(1),
                    'current_plan': None  # Extract from STATE.md if available
                }
        return {'current_phase': '15', 'current_plan': None}

    def load_roadmap_summary(self) -> Dict:
        """Load roadmap summary from ROADMAP.md."""
        # Parse .planning/ROADMAP.md for milestone, goal, progress
        with open('.planning/ROADMAP.md') as f:
            content = f.read()
            # Extract current milestone, phases_complete, next_phase
            return {
                'current_milestone': 'M3: Self-Healing',
                'goal': 'Autonomous remediation + performance optimization',
                'phases_complete': [f'{i}-{j}' for i in range(1, 15) for j in range(1, 4)],
                'next_phase': '16-admin-data-visibility'
            }

    def load_relevant_remedies(
        self,
        failure_context: str,
        limit: int = 5
    ) -> List[RemedyTemplate]:
        """Load relevant remedy templates from PostgreSQL cache or Neo4j.

        Strategy:
        1. Check PostgreSQL cache (fast: <5ms)
        2. If cache miss, query Neo4j graph (slower: 20-50ms)
        3. Update cache on successful Neo4j query

        Relevance scoring:
        - Fingerprint substring match: +0.5
        - Vector similarity (if available): +0.3
        - Recent application (last 30 days): +0.2
        """
        # Try PostgreSQL cache first
        cached = RemedyTemplate.query_relevant(failure_context, limit)
        if cached:
            return cached

        # Fallback to Neo4j
        if not self.graph_client:
            return []

        query = '''
        MATCH (t:RemedyTemplate)
        WHERE t.fingerprint CONTAINS $context OR t.description CONTAINS $context
        RETURN t.fingerprint AS fingerprint,
               t.description AS description,
               t.confidence AS confidence,
               t.application_count AS application_count,
               t.last_applied_at AS last_applied_at
        ORDER BY t.application_count DESC, t.confidence DESC
        LIMIT $limit
        '''
        result = self.graph_client.execute_query(query, {
            'context': failure_context,
            'limit': limit
        })

        # Convert to RemedyTemplate objects and cache
        templates = [
            RemedyTemplate.from_graph_result(record)
            for record in result
        ]

        # Update PostgreSQL cache
        for template in templates:
            template.cache_refresh()

        return templates

    def _fallback_git_log(self, limit: int) -> List[Dict]:
        """Fallback to git log when graph unavailable."""
        import subprocess
        result = subprocess.run(
            ['git', 'log', f'-{limit}', '--pretty=format:%H|%s|%ct', '--name-only'],
            capture_output=True, text=True, cwd='.'
        )
        # Parse output into commit objects
        commits = []
        for line in result.stdout.split('\n\n'):
            if not line.strip():
                continue
            parts = line.split('\n')
            sha, msg, ts = parts[0].split('|')
            changed_files = parts[1:] if len(parts) > 1 else []
            commits.append({
                'sha': sha,
                'message': msg,
                'timestamp': datetime.fromtimestamp(int(ts)),
                'changed_files': changed_files
            })
        return commits
```

from sqlalchemy import Column, Integer, String, Text, Numeric, TIMESTAMP, Index
from sqlalchemy.sql import func
from src.models import db
from typing import List, Optional

class RemedyTemplate(db.Model):
    """PostgreSQL cache for frequently-used remedy templates.

    Source of truth: Neo4j graph
    Cache strategy: Last 30 days, >3 applications
    Sync: Every 5 minutes or on template promotion
    """
    __tablename__ = 'remedy_template_cache'

    id = Column(Integer, primary_key=True)
    template_id = Column(String(36), unique=True, nullable=False)  # UUID from Neo4j

    # Template identity
    fingerprint = Column(String(255), nullable=False, index=True)
    description = Column(Text)
    remedy_payload = Column(Text)  # Serialized fix (JSON or code)

    # Confidence & usage
    confidence = Column(Numeric(3, 2), nullable=False)
    application_count = Column(Integer, default=0)
    success_count = Column(Integer, default=0)

    # Temporal metadata
    last_applied_at = Column(TIMESTAMP)
    cache_refreshed_at = Column(TIMESTAMP, server_default=func.now())
    expires_at = Column(TIMESTAMP)  # Adaptive expiry based on file changes

    # Indexes for fast lookup
    __table_args__ = (
        Index('idx_remedy_fingerprint_conf', 'fingerprint', 'confidence'),
        Index('idx_remedy_last_applied', 'last_applied_at'),
    )

    @classmethod
    def query_relevant(cls, failure_context: str, limit: int = 5) -> List['RemedyTemplate']:
        """Query cache for relevant templates.

        Relevance: fingerprint substring match + recent application.
        """
        from datetime import datetime, timedelta
        recent_threshold = datetime.now() - timedelta(days=30)

        return cls.query.filter(
            cls.fingerprint.contains(failure_context),
            cls.confidence >= 0.7,
            cls.last_applied_at >= recent_threshold
        ).order_by(
            cls.application_count.desc(),
            cls.confidence.desc()
        ).limit(limit).all()

    @classmethod
    def from_graph_result(cls, record: Dict) -> 'RemedyTemplate':
        """Create instance from Neo4j query result."""
        return cls(
            template_id=record.get('template_id'),
            fingerprint=record['fingerprint'],
            description=record['description'],
            confidence=record['confidence'],
            application_count=record.get('application_count', 0),
            last_applied_at=record.get('last_applied_at')
        )

    def cache_refresh(self):
        """Update cache refresh timestamp."""
        self.cache_refreshed_at = func.now()
        db.session.commit()

    def mark_applied(self, success: bool = True):
        """Increment application count and update timestamp."""
        self.application_count += 1
        if success:
            self.success_count += 1
        self.last_applied_at = func.now()
        db.session.commit()
```
