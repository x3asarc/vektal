```python
from typing import Dict
from src.core.graphiti_client import get_graphiti_client
from src.models.remedy_templates import RemedyTemplate
from src.models.sandbox_runs import SandboxRun
import json

class TemplateExtractor:
    def __init__(self):
        self.graph_client = get_graphiti_client()

    def check_promotion_eligibility(self, fix_payload: Dict, fingerprint: str) -> bool:
        """Check if fix should be promoted to template."""
        # Query: Count successful applications of this fix pattern
        query = '''
        MATCH (r:SandboxRun {verdict: 'GREEN'})
        WHERE r.fingerprint = $fingerprint
        RETURN count(r) AS success_count
        '''
        result = self.graph_client.execute_query(query, {'fingerprint': fingerprint})
        success_count = result[0]['success_count'] if result else 0

        # Promotion threshold: ≥2 successful applications
        return success_count >= 2

    def extract_and_promote(self, fix_payload: Dict, confidence: float, fingerprint: str):
        """Promote fix to template in Neo4j + PostgreSQL cache."""
        # Create Neo4j template node
        template_query = '''
        CREATE (t:RemedyTemplate {
            template_id: randomUUID(),
            fingerprint: $fingerprint,
            description: $description,
            remedy_payload: $payload,
            confidence: $confidence,
            application_count: 2,
            created_at: datetime()
        })
        RETURN t.template_id AS template_id
        '''
        result = self.graph_client.execute_query(template_query, {
            'fingerprint': fingerprint,
            'description': fix_payload.get('description', ''),
            'payload': json.dumps(fix_payload['changed_files']),
            'confidence': confidence
        })

        template_id = result[0]['template_id']

        # Sync to PostgreSQL cache
        template = RemedyTemplate(
            template_id=template_id,
            fingerprint=fingerprint,
            description=fix_payload.get('description'),
            remedy_payload=json.dumps(fix_payload['changed_files']),
            confidence=confidence,
            application_count=2,
            success_count=2
        )
        from src.models import db
        db.session.add(template)
        db.session.commit()

        return template_id
```

CLI for manual template promotion and automated sync:

```python
#!/usr/bin/env python
import click
from src.graph.template_extractor import TemplateExtractor

@click.group()
def cli():
    """Template management CLI."""
    pass

@cli.command()
@click.option('--fingerprint', required=True)
@click.option('--sandbox-run-id', required=True)
def promote(fingerprint, sandbox_run_id):
    """Manually promote sandbox run to template."""
    from src.models.sandbox_runs import SandboxRun
    run = SandboxRun.query.filter_by(run_id=sandbox_run_id).first()
    if not run or run.verdict != 'GREEN':
        click.echo("ERROR: Run not found or not GREEN")
        return

    extractor = TemplateExtractor()
    template_id = extractor.extract_and_promote(
        fix_payload={'changed_files': run.changed_files, 'description': ''},
        confidence=run.confidence,
        fingerprint=fingerprint
    )
    click.echo(f"✓ Promoted to template: {template_id}")

@cli.command()
def sync_cache():
    """Sync Neo4j templates to PostgreSQL cache."""
    from src.models.remedy_templates import RemedyTemplate
    from src.core.graphiti_client import get_graphiti_client

    graph = get_graphiti_client()
    query = '''
    MATCH (t:RemedyTemplate)
    WHERE t.application_count >= 3
      AND t.last_applied_at > datetime() - duration({days: 30})
    RETURN t
    '''
    templates = graph.execute_query(query, {})

    # Update PostgreSQL cache
    for t in templates:
        RemedyTemplate.upsert_from_graph(t)

    click.echo(f"✓ Synced {len(templates)} templates to cache")

if __name__ == '__main__':
    cli()
```
