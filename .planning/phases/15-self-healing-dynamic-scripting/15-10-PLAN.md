```python
from typing import Dict, List
from datetime import datetime, timedelta
from src.graph.template_extractor import TemplateExtractor
import logging

logger = logging.getLogger(__name__)

class SentryFeedbackLoop:
    """Validates remediation efficacy by checking Sentry issue state."""

    def __init__(self, sentry_client):
        self.sentry_client = sentry_client
        self.template_extractor = TemplateExtractor()

    def validate_pending_remediations(self) -> List[Dict]:
        """Check all pending remediations from last 24h."""
        # Query database for remediations applied in last 24h
        from src.models.sandbox_runs import SandboxRun
        cutoff = datetime.now() - timedelta(hours=24)

        pending_runs = SandboxRun.query.filter(
            SandboxRun.completed_at >= cutoff,
            SandboxRun.verdict == 'GREEN'
        ).all()

        results = []
        for run in pending_runs:
            if not run.failure_fingerprint:
                continue

            # Extract Sentry issue ID from fingerprint
            sentry_issue_id = self._extract_issue_id(run.failure_fingerprint)
            if not sentry_issue_id:
                continue

            # Query Sentry for current issue state
            issue = self.sentry_client.get_issue(sentry_issue_id)

            # Validate: Issue should be resolved or have no new events
            is_resolved = issue['status'] == 'resolved'
            has_new_events = len(issue.get('activity', [])) > 0

            if is_resolved and not has_new_events:
                # Success - promote to template
                logger.info(f"Remediation successful for {sentry_issue_id}")
                self._promote_to_template(run)
                results.append({
                    'run_id': run.run_id,
                    'status': 'validated',
                    'sentry_issue_id': sentry_issue_id
                })
            elif has_new_events:
                # Failure - same error recurring
                logger.warning(f"Remediation failed for {sentry_issue_id} - issue recurring")
                self._mark_failed(run)
                results.append({
                    'run_id': run.run_id,
                    'status': 'failed',
                    'sentry_issue_id': sentry_issue_id,
                    'reason': 'issue_recurring'
                })
            else:
                # Pending - wait longer
                results.append({
                    'run_id': run.run_id,
                    'status': 'pending',
                    'sentry_issue_id': sentry_issue_id
                })

        return results

    def _promote_to_template(self, run: SandboxRun):
        """Promote validated fix to template library."""
        template_id = self.template_extractor.extract_and_promote(
            fix_payload={'changed_files': run.changed_files},
            confidence=run.confidence,
            fingerprint=run.failure_fingerprint
        )
        logger.info(f"Promoted run {run.run_id} to template {template_id}")

    def _mark_failed(self, run: SandboxRun):
        """Mark remediation as failed."""
        run.rollback_notes = f"Failed validation - issue recurring after remediation"
        from src.models import db
        db.session.commit()

    def _extract_issue_id(self, fingerprint: str) -> str:
        """Extract Sentry issue ID from fingerprint."""
        # Fingerprint format: "SENTRY-123:src/module.py:ErrorType"
        if fingerprint.startswith('SENTRY-'):
            return fingerprint.split(':')[0]
        return None
```

```python
#!/usr/bin/env python
import click
from src.graph.sentry_feedback_loop import SentryFeedbackLoop
from sentry_sdk import Hub

@click.command()
@click.option('--hours', default=24, help='Validate last N hours')
def validate(hours):
    """Validate remediation efficacy."""
    client = Hub.current.client  # Sentry client
    loop = SentryFeedbackLoop(client)

    results = loop.validate_pending_remediations()

    click.echo(f"Validated {len(results)} remediations:")
    validated = [r for r in results if r['status'] == 'validated']
    failed = [r for r in results if r['status'] == 'failed']
    pending = [r for r in results if r['status'] == 'pending']

    click.echo(f"  ✓ Validated: {len(validated)}")
    click.echo(f"  ✗ Failed: {len(failed)}")
    click.echo(f"  ⏳ Pending: {len(pending)}")

    if failed:
        click.echo("\nFailed remediations:")
        for f in failed:
            click.echo(f"  - {f['sentry_issue_id']}: {f['reason']}")

if __name__ == '__main__':
    validate()
```
