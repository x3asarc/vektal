## Goal

Add **context window optimization** through:
1. `compact_output: bool` parameter on `query_graph` tool - strips verbose fields
2. Edge-type score multipliers in `search_expand_bridge.py` - structural ranking
3. Temporal decay on embeddings - prioritize recent nodes

**Token savings target**: 40% reduction on large graph retrievals (>10 nodes)

---

## Background

**Tool Calling 2.0 Pattern**: "Context Window Optimization" - Filter noise before entering LLM context. Run data transformations server-side, not in LLM inference.

**Current bottleneck**:
- Graph nodes return full description fields (~300 tokens/node)
- Only ~10-15 tokens contribute actual signal
- 8192-token budget exhausted by ~16 nodes
- Verbose output blocks ability to retrieve more nodes

**After compact mode**:
- Return `{path, type, summary}` only (~30 tokens/node)
- 8192-token budget accommodates ~270 nodes (17× increase)
- Actual capacity: ~50 useful nodes (vs 16 previously = 3.1× improvement)

**Source**: [Anthropic Context Window Optimization](https://platform.claude.com/docs/en/agents-and-tools/tool-use/programmatic-tool-calling) - "Return only parts that matter to the model"

```python
# src/graph/search_expand_bridge.py

def serialize_node(node: dict, compact: bool = False) -> dict:
    """Serialize graph node. Compact mode strips verbose fields."""
    if compact:
        return {
            "path": node.get("path", node.get("name")),
            "type": node.get("type", node.get("label")),
            "summary": _generate_summary(node)  # First 100 chars of description
        }

    # Full mode: return all fields
    return node


def _generate_summary(node: dict) -> str:
    """Generate compact summary for node."""
    desc = node.get("description", "")
    if len(desc) > 100:
        return desc[:97] + "..."
    return desc


def format_search_results(nodes: list[dict], compact: bool = False) -> dict:
    """Format graph results with optional compact mode."""
    return {
        "nodes": [serialize_node(n, compact) for n in nodes],
        "total": len(nodes),
        "mode": "compact" if compact else "full",
        "token_estimate": len(nodes) * (30 if compact else 50)
    }
```

### Step 4: Add Edge-Type Score Multipliers

```python
# src/graph/search_expand_bridge.py

EDGE_SCORE_MULTIPLIER = {
    "IMPLEMENTS": 1.3,     # Strong: actual implementation
    "CALLS": 1.1,          # Medium: direct function call
    "IMPORTS": 0.9,        # Medium-weak: dependency
    "REFERENCES": 0.7,     # Weak: documentation mention
    "EXPLAINS": 1.0,       # Neutral: planning doc link
    "DEPENDS_ON": 0.9,     # Similar to IMPORTS
    "CONTAINS": 1.0        # Structural: module contains class
}


def score_node_with_edge_type(
    base_score: float,
    edge_type: str,
    anchor_score: float = 1.0
) -> float:
    """Apply edge-type multiplier to node score."""
    multiplier = EDGE_SCORE_MULTIPLIER.get(edge_type, 1.0)
    return base_score * multiplier * anchor_score


def expand_from_anchors(anchors: list[dict], depth: int, compact: bool) -> dict:
    """BFS expansion with edge-type scoring."""
    visited = set()
    results = []

    for anchor in anchors:
        anchor_id = anchor["id"]
        anchor_score = anchor.get("score", 1.0)

        # BFS traversal
        queue = [(anchor_id, 0, anchor_score)]  # (node_id, current_depth, parent_score)

        while queue:
            node_id, current_depth, parent_score = queue.pop(0)

            if node_id in visited or current_depth > depth:
                continue

            visited.add(node_id)

            # Get node + its edges
            node, edges = get_node_with_edges(node_id)

            # Apply edge-type scoring
            for edge in edges:
                neighbor_id = edge["target"]
                edge_type = edge["type"]

                neighbor_score = score_node_with_edge_type(
                    base_score=parent_score,
                    edge_type=edge_type,
                    anchor_score=anchor_score
                )

                queue.append((neighbor_id, current_depth + 1, neighbor_score))
                results.append({
                    **node,
                    "score": neighbor_score,
                    "edge_type": edge_type,
                    "depth": current_depth + 1
                })

    return format_search_results(results, compact)
```

### Step 5: Add Temporal Decay

```python
# src/graph/search_expand_bridge.py

import math
from datetime import datetime

def apply_temporal_decay(score: float, created_at: str) -> float:
    """Apply temporal decay to score. Nodes >6 months score 0.6× base."""
    if not created_at:
        return score

    try:
        created = datetime.fromisoformat(created_at)
        age_days = (datetime.utcnow() - created).days

        # Exponential decay: exp(-0.01 × age_days)
        # 180 days (6 months) → 0.6× multiplier
        # 365 days (1 year) → 0.36× multiplier
        decay_factor = math.exp(-0.01 * age_days)

        # Floor at 0.5 (50% of base score)
        return score * max(decay_factor, 0.5)

    except (ValueError, TypeError):
        return score  # Invalid timestamp, no decay


def score_node_with_edge_type(
    base_score: float,
    edge_type: str,
    anchor_score: float = 1.0,
    created_at: str | None = None  # NEW
) -> float:
    """Apply edge-type multiplier + temporal decay."""
    multiplier = EDGE_SCORE_MULTIPLIER.get(edge_type, 1.0)
    score = base_score * multiplier * anchor_score

    # Apply temporal decay
    if created_at:
        score = apply_temporal_decay(score, created_at)

    return score
```

```python
# tests/graph/test_compact_output.py

def test_compact_mode_reduces_tokens():
    """Verify compact mode uses ≤35 tokens/node."""
    from src.graph.search_expand_bridge import serialize_node

    node = {
        "path": "src/core/graphiti_client.py",
        "type": "File",
        "description": "A" * 300  # 300-char description
    }

    # Full mode
    full = serialize_node(node, compact=False)
    full_tokens = len(str(full))
    assert full_tokens > 300  # Full description included

    # Compact mode
    compact = serialize_node(node, compact=True)
    compact_tokens = len(str(compact))
    assert compact_tokens < 150  # Only summary
    assert "summary" in compact
    assert len(compact["summary"]) <= 100


def test_edge_type_multipliers():
    """Verify edge types affect scoring."""
    from src.graph.search_expand_bridge import score_node_with_edge_type

    base_score = 1.0

    implements_score = score_node_with_edge_type(base_score, "IMPLEMENTS")
    references_score = score_node_with_edge_type(base_score, "REFERENCES")

    assert implements_score > base_score  # 1.3×
    assert references_score < base_score  # 0.7×
    assert implements_score > references_score


def test_temporal_decay():
    """Verify old nodes score lower."""
    from src.graph.search_expand_bridge import apply_temporal_decay
    from datetime import timedelta, datetime

    base_score = 1.0

    # Fresh node (today)
    fresh = apply_temporal_decay(base_score, datetime.utcnow().isoformat())
    assert fresh == base_score  # No decay

    # 6-month-old node
    old_date = (datetime.utcnow() - timedelta(days=180)).isoformat()
    old_score = apply_temporal_decay(base_score, old_date)
    assert 0.5 < old_score < 0.7  # ~0.6× multiplier
```

```python
# tests/integration/test_token_reduction.py

def test_compact_output_token_savings():
    """Measure token reduction on 20-node query."""
    from src.graph.mcp_server import call_tool
    import json

    query = "find all files importing graphiti_client"

    # Full mode
    full_result = call_tool("query_graph", {
        "query": query,
        "compact_output": False
    })
    full_tokens = len(json.dumps(full_result))

    # Compact mode
    compact_result = call_tool("query_graph", {
        "query": query,
        "compact_output": True
    })
    compact_tokens = len(json.dumps(compact_result))

    # Assert 20% reduction minimum
    savings = (full_tokens - compact_tokens) / full_tokens
    assert savings >= 0.20, f"Only {savings:.1%} savings, expected ≥20%"
```
