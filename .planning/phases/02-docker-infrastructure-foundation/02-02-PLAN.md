Create docker-compose.yml in project root with all 6 services.

**Services to define:**
1. **nginx** - Reverse proxy (port 80)
2. **backend** - Flask API (port 5000)
3. **frontend** - Next.js (port 3000)
4. **celery_worker** - Background job processor
5. **db** - PostgreSQL (port 5432)
6. **redis** - Cache and broker (port 6379)

**docker-compose.yml content:**
```yaml
# Shopify Multi-Supplier Platform - Docker Compose
# Development configuration with hot reload and debug ports
#
# Usage:
#   docker compose up        # Start all services
#   docker compose up -d     # Start in background
#   docker compose down      # Stop all services
#   docker compose logs -f   # View logs (all services)
#   docker compose logs -f backend  # View specific service logs

services:
  # ===========================================
  # Nginx Reverse Proxy
  # Primary entry point: http://localhost
  # ===========================================
  nginx:
    image: nginx:1.25
    ports:
      - "${NGINX_PORT:-80}:80"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
    depends_on:
      backend:
        condition: service_started
      # frontend dependency added when Next.js is ready (Phase 7)
    restart: unless-stopped
    networks:
      - app-network

  # ===========================================
  # Flask Backend API
  # Direct access: http://localhost:5000 (debug only)
  # ===========================================
  backend:
    build:
      context: .
      dockerfile: Dockerfile.backend
    ports:
      - "${FLASK_PORT:-5000}:5000"
    volumes:
      # Bind mounts for hot reload
      - ./src:/app/src
      - ./utils:/app/utils
      - ./config:/app/config
      # Persistent volumes
      - backend_uploads:/app/uploads
      - backend_data:/app/data
    environment:
      - FLASK_ENV=${FLASK_ENV:-development}
      - FLASK_DEBUG=${FLASK_DEBUG:-1}
      - FLASK_APP=src/app.py
      - FLASK_SECRET_KEY=${FLASK_SECRET_KEY}
      - DATABASE_URL=postgresql://${DB_USER:-admin}:${DB_PASSWORD}@db:5432/${DB_NAME:-shopify_platform}
      - REDIS_URL=redis://redis:6379/0
      - CELERY_BROKER_URL=redis://redis:6379/0
      # AI service keys
      - GEMINI_API_KEY=${GEMINI_API_KEY}
      - OPENROUTER_API_KEY=${OPENROUTER_API_KEY}
      - VISION_AI_PROVIDER=${VISION_AI_PROVIDER:-openrouter}
      - VISION_AI_MODEL=${VISION_AI_MODEL:-google/gemini-flash-1.5-8b}
      # Shopify credentials
      - SHOPIFY_API_KEY=${SHOPIFY_API_KEY}
      - SHOPIFY_API_SECRET=${SHOPIFY_API_SECRET}
      - API_VERSION=${API_VERSION:-2024-01}
    depends_on:
      db:
        condition: service_healthy
      redis:
        condition: service_healthy
    restart: unless-stopped
    networks:
      - app-network

  # ===========================================
  # Celery Worker (Background Jobs)
  # No direct port - processes jobs from Redis queue
  # ===========================================
  celery_worker:
    build:
      context: .
      dockerfile: Dockerfile.backend
    volumes:
      - ./src:/app/src
      - ./utils:/app/utils
      - ./config:/app/config
      - backend_uploads:/app/uploads
      - backend_data:/app/data
    environment:
      - DATABASE_URL=postgresql://${DB_USER:-admin}:${DB_PASSWORD}@db:5432/${DB_NAME:-shopify_platform}
      - CELERY_BROKER_URL=redis://redis:6379/0
      - CELERY_RESULT_BACKEND=redis://redis:6379/0
      # AI service keys (for scraping tasks)
      - GEMINI_API_KEY=${GEMINI_API_KEY}
      - OPENROUTER_API_KEY=${OPENROUTER_API_KEY}
      - VISION_AI_PROVIDER=${VISION_AI_PROVIDER:-openrouter}
      - VISION_AI_MODEL=${VISION_AI_MODEL:-google/gemini-flash-1.5-8b}
      # Shopify credentials (for API tasks)
      - SHOPIFY_API_KEY=${SHOPIFY_API_KEY}
      - SHOPIFY_API_SECRET=${SHOPIFY_API_SECRET}
      - API_VERSION=${API_VERSION:-2024-01}
    command: celery -A src.celery_app worker --loglevel=info --concurrency=2
    depends_on:
      db:
        condition: service_healthy
      redis:
        condition: service_healthy
    restart: unless-stopped
    networks:
      - app-network

  # ===========================================
  # Next.js Frontend
  # Direct access: http://localhost:3000 (debug only)
  # Full implementation in Phase 7
  # ===========================================
  frontend:
    image: node:20-slim
    ports:
      - "${NEXTJS_PORT:-3000}:3000"
    volumes:
      - ./frontend:/app
      # Prevent overwriting node_modules and .next
      - /app/node_modules
      - /app/.next
    working_dir: /app
    environment:
      - NODE_ENV=development
      - WATCHPACK_POLLING=true
      - NEXT_PUBLIC_API_URL=http://localhost/api
    # Install dependencies on first start, then run dev server
    command: sh -c "npm install && npm run dev"
    restart: unless-stopped
    networks:
      - app-network

  # ===========================================
  # PostgreSQL Database
  # Direct access: localhost:5432 (debug only)
  # Data persists via named volume
  # ===========================================
  db:
    image: postgres:15
    ports:
      - "${POSTGRES_PORT:-5432}:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
      # Init scripts run on first start (Phase 3)
      # - ./db/init:/docker-entrypoint-initdb.d
    environment:
      - POSTGRES_DB=${DB_NAME:-shopify_platform}
      - POSTGRES_USER=${DB_USER:-admin}
      - POSTGRES_PASSWORD=${DB_PASSWORD}
    # Startup ordering health check - NOT automatic recovery monitoring (Phase 13)
    healthcheck:
      test: ["CMD", "pg_isready", "-U", "${DB_USER:-admin}", "-d", "${DB_NAME:-shopify_platform}"]
      interval: 5s
      timeout: 5s
      retries: 5
      start_period: 10s
    restart: unless-stopped
    networks:
      - app-network

  # ===========================================
  # Redis (Cache + Celery Broker)
  # Direct access: localhost:6379 (debug only)
  # ===========================================
  redis:
    image: redis:7-alpine
    ports:
      - "${REDIS_PORT:-6379}:6379"
    volumes:
      - redis_data:/data
    # Startup ordering health check - NOT automatic recovery monitoring (Phase 13)
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 5s
      timeout: 5s
      retries: 5
    restart: unless-stopped
    networks:
      - app-network

# ===========================================
# Named Volumes (persistent data)
# ===========================================
volumes:
  postgres_data:
    name: shopify_postgres_data
  redis_data:
    name: shopify_redis_data
  backend_uploads:
    name: shopify_backend_uploads
  backend_data:
    name: shopify_backend_data

# ===========================================
# Networks
# ===========================================
networks:
  app-network:
    driver: bridge
```

**Key design decisions:**
- Health checks ensure services start only when dependencies are ready (startup ordering, NOT production monitoring)
- Bind mounts (./src:/app/src) enable hot reload without container rebuild
- Named volumes persist database data across `docker compose down`
- All debug ports exposed (per CONTEXT.md decision for learning phase)
- Environment variable substitution from .env file
- Celery worker concurrency=2 (conservative for development)
- Frontend runs `npm install && npm run dev` to auto-setup on first start

**nginx/nginx.conf content:**
```nginx
# Nginx Reverse Proxy Configuration
# Shopify Multi-Supplier Platform - Development Mode
#
# Routing:
#   /api/*        -> Flask backend (port 5000)
#   /socket.io/*  -> Flask backend (WebSocket - Phase 9-10)
#   /*            -> Next.js frontend (port 3000)

events {
    worker_connections 1024;
}

http {
    # Include MIME types
    include /etc/nginx/mime.types;
    default_type application/octet-stream;

    # Logging - quiet by default, enable for debugging
    # access_log /var/log/nginx/access.log;
    access_log off;
    error_log /var/log/nginx/error.log warn;

    # Upstream definitions for clarity
    upstream backend {
        server backend:5000;
    }

    upstream frontend {
        server frontend:3000;
    }

    server {
        listen 80;
        server_name localhost;

        # Increase timeouts for long-running requests (AI analysis)
        proxy_connect_timeout 60s;
        proxy_send_timeout 120s;
        proxy_read_timeout 120s;

        # -----------------------------------------
        # Flask API Routes
        # -----------------------------------------
        location /api/ {
            proxy_pass http://backend/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;

            # Disable buffering for streaming responses
            proxy_buffering off;
        }

        # Auth routes (Shopify OAuth)
        location /auth/ {
            proxy_pass http://backend/auth/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # Webhook routes
        location /webhooks/ {
            proxy_pass http://backend/webhooks/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # Health check endpoint
        location /health {
            proxy_pass http://backend/health;
            proxy_set_header Host $host;
        }

        # -----------------------------------------
        # WebSocket Support (Phase 9-10)
        # Socket.IO requires special headers for upgrade
        # -----------------------------------------
        location /socket.io/ {
            proxy_pass http://backend/socket.io/;
            proxy_http_version 1.1;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection "Upgrade";
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;

            # WebSocket connections can be long-lived
            proxy_read_timeout 86400s;
            proxy_buffering off;
        }

        # -----------------------------------------
        # Next.js Frontend (Default Route)
        # -----------------------------------------
        location / {
            proxy_pass http://frontend;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;

            # Next.js hot reload requires WebSocket upgrade
            proxy_http_version 1.1;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection "Upgrade";
        }

        # -----------------------------------------
        # Static file caching (production - Phase 13)
        # -----------------------------------------
        # location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2)$ {
        #     proxy_pass http://frontend;
        #     expires 7d;
        #     add_header Cache-Control "public, immutable";
        # }
    }
}
```

**Routing logic explained:**
- `/api/*` strips `/api` prefix and forwards to Flask (so `/api/jobs` becomes `/jobs` on backend)
- `/auth/*` and `/webhooks/*` preserved (Shopify OAuth expects exact paths)
- `/socket.io/*` has special WebSocket headers for real-time features (Phase 9-10)
- `/*` (default) goes to Next.js frontend
- 120s read timeout accommodates Vision AI and scraping operations

**Why access_log off:** Development mode prioritizes clean console output. Enable by uncommenting the access_log line when debugging.

```python
"""
Celery Application Configuration
Shopify Multi-Supplier Platform

This is a minimal placeholder for Phase 2 Docker setup.
Full implementation with tasks added in Phase 6.

Usage:
    # Run worker (from docker-compose)
    celery -A src.celery_app worker --loglevel=info

    # Run worker locally (for debugging)
    celery -A src.celery_app worker --loglevel=debug --concurrency=1
"""
import os
from celery import Celery

# Create Celery app
app = Celery('shopify_platform')

# Configuration from environment variables
app.conf.update(
    broker_url=os.getenv('CELERY_BROKER_URL', 'redis://localhost:6379/0'),
    result_backend=os.getenv('CELERY_RESULT_BACKEND', 'redis://localhost:6379/0'),

    # Task settings
    task_serializer='json',
    accept_content=['json'],
    result_serializer='json',
    timezone='UTC',
    enable_utc=True,

    # Worker settings
    worker_prefetch_multiplier=1,  # Fair task distribution
    task_acks_late=True,           # Retry on worker crash

    # Result expiration (1 hour)
    result_expires=3600,
)


# Example task (placeholder - real tasks in Phase 6)
@app.task(bind=True)
def debug_task(self):
    """Test task to verify Celery is working."""
    print(f'Request: {self.request!r}')
    return {'status': 'ok', 'message': 'Celery is working!'}


# Task discovery - will find tasks in src.tasks module (Phase 6)
# app.autodiscover_tasks(['src.tasks'])
```

Why this placeholder:
- docker-compose.yml references `src.celery_app` for the worker command
- Without this file, celery_worker service crashes immediately
- The debug_task allows testing Celery connectivity: `docker compose exec backend celery -A src.celery_app call src.celery_app.debug_task`
