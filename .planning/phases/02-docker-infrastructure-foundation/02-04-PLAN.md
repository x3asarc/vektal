   - Falls back to `os.getenv(secret_name)` for local development
   - Optionally accepts a default value
   - Strips whitespace from file contents (Docker secrets often have trailing newlines)
   - Logs which source was used (file vs env) at DEBUG level for troubleshooting

```python
"""
Docker secrets utility with environment variable fallback.

In Docker Swarm/Compose with secrets, sensitive values are mounted as files
at /run/secrets/{secret_name}. This module provides a unified interface
that works both in Docker (reads files) and local development (reads env vars).

Usage:
    from src.core.secrets import get_secret

    api_key = get_secret("GEMINI_API_KEY")
    db_password = get_secret("DB_PASSWORD", default="dev-password")
"""

import os
import logging

logger = logging.getLogger(__name__)

SECRETS_DIR = "/run/secrets"


def get_secret(name: str, default: str = None) -> str | None:
    """
    Get a secret value from Docker secrets file or environment variable.

    Priority:
    1. Docker secret file at /run/secrets/{name}
    2. Environment variable {name}
    3. Default value (if provided)

    Args:
        name: Secret name (e.g., "GEMINI_API_KEY")
        default: Default value if secret not found

    Returns:
        Secret value or default
    """
    secret_path = os.path.join(SECRETS_DIR, name)

    # Try Docker secret file first
    if os.path.isfile(secret_path):
        try:
            with open(secret_path, 'r') as f:
                value = f.read().strip()
                logger.debug(f"Secret {name} loaded from Docker secrets file")
                return value
        except (IOError, PermissionError) as e:
            logger.warning(f"Could not read secret file {secret_path}: {e}")

    # Fall back to environment variable
    env_value = os.getenv(name)
    if env_value is not None:
        logger.debug(f"Secret {name} loaded from environment variable")
        return env_value

    # Return default
    if default is not None:
        logger.debug(f"Secret {name} using default value")
    return default
```

```yaml
# Docker Compose secrets overlay
# Usage: docker compose -f docker-compose.yml -f docker-compose.secrets.yml up
#
# This overlay replaces environment variable secrets with file-based Docker secrets.
# Secrets are mounted at /run/secrets/{secret_name} inside containers.
#
# Setup:
# 1. Create secrets/ directory (already exists via .gitkeep)
# 2. Create secret files: echo "your-api-key" > secrets/GEMINI_API_KEY
# 3. Run with: docker compose -f docker-compose.yml -f docker-compose.secrets.yml up

secrets:
  # Database credentials
  DB_PASSWORD:
    file: ./secrets/DB_PASSWORD

  # Flask session key
  FLASK_SECRET_KEY:
    file: ./secrets/FLASK_SECRET_KEY

  # AI service keys
  GEMINI_API_KEY:
    file: ./secrets/GEMINI_API_KEY
  OPENROUTER_API_KEY:
    file: ./secrets/OPENROUTER_API_KEY

  # Shopify credentials
  SHOPIFY_API_KEY:
    file: ./secrets/SHOPIFY_API_KEY
  SHOPIFY_API_SECRET:
    file: ./secrets/SHOPIFY_API_SECRET

services:
  backend:
    secrets:
      - DB_PASSWORD
      - FLASK_SECRET_KEY
      - GEMINI_API_KEY
      - OPENROUTER_API_KEY
      - SHOPIFY_API_KEY
      - SHOPIFY_API_SECRET
    environment:
      # Override sensitive vars to empty - app reads from /run/secrets/
      - FLASK_SECRET_KEY=
      - GEMINI_API_KEY=
      - OPENROUTER_API_KEY=
      - SHOPIFY_API_KEY=
      - SHOPIFY_API_SECRET=
      # DB_PASSWORD handled specially in DATABASE_URL construction
      - DATABASE_URL=postgresql://${DB_USER:-admin}:@db:5432/${DB_NAME:-shopify_platform}

  celery_worker:
    secrets:
      - DB_PASSWORD
      - GEMINI_API_KEY
      - OPENROUTER_API_KEY
      - SHOPIFY_API_KEY
      - SHOPIFY_API_SECRET
    environment:
      # Override sensitive vars to empty - app reads from /run/secrets/
      - GEMINI_API_KEY=
      - OPENROUTER_API_KEY=
      - SHOPIFY_API_KEY=
      - SHOPIFY_API_SECRET=
      - DATABASE_URL=postgresql://${DB_USER:-admin}:@db:5432/${DB_NAME:-shopify_platform}

  db:
    secrets:
      - DB_PASSWORD
    environment:
      # PostgreSQL reads POSTGRES_PASSWORD_FILE for file-based secrets
      - POSTGRES_PASSWORD_FILE=/run/secrets/DB_PASSWORD
      - POSTGRES_PASSWORD=
```
