    NOTE: This class does NOT load sentence-transformers. Embeddings are
    generated by EmbeddingGenerator (Plan 04) and passed to find_similar_products().
    This prevents duplicate model loading and ensures single source of truth.
    """

    # Cost estimates per 1K products

        'openai/gpt-4o-mini': '~$0.05',

    }

    def __init__(self,
                 api_key: str = None,
                 model: str = "google/gemini-flash-1.5",
                 site_name: str = "Bastelschachtel"):
        self.api_key = api_key or os.getenv('OPENROUTER_API_KEY')
        self.model = model
        self.site_name = site_name

        # Cache for AI responses (30 day TTL)
        self.cache = TTLCache(maxsize=10000, ttl=2592000)

        # NOTE: No embedding model here - embeddings come from EmbeddingGenerator

    def find_similar_products(self, target_embedding: np.ndarray,
                             catalog_embeddings: np.ndarray,
                             catalog_data: list,
                             top_k: int = 3) -> list:
        """Find similar products for few-shot examples.

        Args:
            target_embedding: Pre-computed 768-dim embedding for target product
                             (from EmbeddingGenerator)
            catalog_embeddings: Pre-computed embeddings for catalog products
            catalog_data: List of product dicts corresponding to embeddings
            top_k: Number of similar products to return

        Returns:
            List of top_k most similar products from catalog
        """
        # IMPLEMENTATION: Complete cosine similarity computation and top_k selection
        # Step 1: Compute cosine similarity between target and all catalog embeddings
        # cosine_similarity returns shape (1, n_catalog), so [0] gets the 1D array
        similarities = cosine_similarity([target_embedding], catalog_embeddings)[0]

        # Step 2: Get indices of top_k highest similarity scores
        # np.argsort returns indices in ascending order, so we take last k and reverse
        top_indices = np.argsort(similarities)[-top_k:][::-1]

        # Step 3: Return the corresponding catalog products
        return [catalog_data[i] for i in top_indices]
